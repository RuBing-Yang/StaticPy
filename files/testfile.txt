def func1() -> None:
    i:int = 1 + 1 * 1 - 1
    if i == 1:
        print("i is ", i)

def func2(l : List) -> List:
    l[1] = 5
    return l

def main() -> int:
    func1()
    l:List = [1, 2, 3]
    l2:List = func2(l)
    i:int = 0
    while i < 3:
        print("list [", i, "] = ", l2[i])
        i = i + 1
    d:Dict = {1:11, 2:22}
    i = 1
    while i < 10:
        print("dict [", i, "] = ", d[i])
        i = i + 1
        if i >= 3:
            break
    return 0
